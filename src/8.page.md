> 分页例1

```
{
  search(query: "graphql", type: REPOSITORY, first: 10) {
    codeCount
    edges {
      node {
        ... on Repository {
          nameWithOwner
          description
          updatedAt
          stargazers {
            totalCount
          }
          forks {
            totalCount
          }
          issues(states: OPEN, first: 3) {
            edges {
              node {
                title
              }
            }
          }
        }
      }
      cursor
    }
    pageInfo {
      startCursor
      endCursor
      hasNextPage
      hasPreviousPage
    }
  }
  rateLimit {
    limit
    cost
    remaining
    resetAt
  }
}
```

> 分页例2

```
{
  search(first: 10, query: "graphql", type: REPOSITORY) {
    codeCount
    issueCount
    nodes {
      # http://spec.graphql.cn/#sec-Inline-Fragments-
      ... on Repository {
        nameWithOwner
        description
        issues(first: 3) {
          nodes {
            author {
              avatarUrl
              login
              resourcePath
              url
            }
          }
        }
      }
    }
  }
}
```


如果想查下一页，直接修改query search，添加after参数。值就是cursor，如 search(first: 3, after:"Y3Vyc29yOjM=", query: "graphql", type: REPOSITORY) {
